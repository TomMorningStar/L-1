      // Подсчитать максимальный объем данных, который можно записать в localStorage вашего браузера.

      function getLocalStorageSize() {
        let data = 'a';
        let size = 0;
        try {
          while (true) {
            localStorage.setItem(data, data);
            data += data;
            size += data.length;
          }
        } catch (e) {
          localStorage.removeItem(data);
        }
        return size;
      }
